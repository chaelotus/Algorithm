function solution(tickets) {
  let answer = [];
  const result = [];

  const withICN = [];
  const withoutICN = [];
  const visited = [];

  for (let i = 0; i < tickets.length; i++) {
    if (tickets[i][0] === 'ICN') {
      withICN.push(tickets[i]);
    } else {
      withoutICN.push(tickets[i]);
    }
  }
  tickets = [...withICN.sort(), ...withoutICN.sort()];

      return true;
